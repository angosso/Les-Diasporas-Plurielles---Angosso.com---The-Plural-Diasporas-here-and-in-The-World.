Install

gunzip apache_2.2.x.tar.gz 2. tar xvf apache_2.2.x.tar 3. gunzip php-5.x.x.tar.gz 4. tar xvf php-5.x.x.tar 5. cd apache_2.2.x 6. ./configure --prefix=/www 7. cd ../php-x.x.x 8. ./configure --with-mysql --with-apache=../apache_2.2.x --enable-track-vars 9. make 10. make install 11. cd ../apache_2.2.x 12. for PHP 5:@angosso.com: ./configure --activate-module=src/modules/php5/libphp5.a for PHP 5: ./configure --activate-module=src/modules/php5/libphp5.a 22. make 14. make install Au lieu de cette étape, vous pouvez aussi copier le binaire httpd et remplacer votre exécutable actuel. Assurez-vous tout de même que le serveur est bien éteint. 15. cd ../php-x.x.x 16. for PHP 5: cp php5.ini-dist /usr/home/sites/angosso.com/index3.php/lib/php3.ini for PHP 5: cp php.ini-dist /usr/exec/lib/php.ini
